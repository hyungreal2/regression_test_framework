	cv = geGetWindowCellView()

    propType = case(type(propValue)
        (fixnum  "int")

procedure(iAddInstProperty(cv propName propType propValue visited cnt)
let(( tmpSCId masterName )
	visited[cv~>cellName] = t
	foreach( inst cv~>instances
		if( dbReplaceProp(inst propName propType propValue) then
			cnt++
		)
		masterName = inst~>cellName
		if( !visited[masterName] then
			visited[masterName] = t
			tmpSCId = dbOpenCellViewByType(inst~>libName inst~>cellName "schematic" nil "a")
			if( tmpSCId then
				cnt = iAddInstProperty( tmpSCId propName propType propValue visited cnt )
			)
		)
	)	
	dbSave(cv)
	dbClose(cv)
	cnt
))

procedure( listHierLibCells(libName cellName @optional (viewName "schematic"))
  let( (seen pairs)
    seen  = makeTable('seen nil)  ; set by "lib/cell" key
    pairs = nil
    procedure( _walk(l c)
      let( (cv key)
        key = sprintf(nil "%s/%s" l c)
        unless(seen[key]
          seen[key] = t
          pairs     = cons(key pairs)
          cv = dbOpenCellViewByType(l c "schematic" "" "r")
          when(cv
            foreach(inst cv~>instances
              when(inst~>master
                _walk(inst~>master~>libName inst~>master~>cellName)
              )
            )
            dbClose(cv)
          )
        )
      )
    )
    _walk(libName cellName)
    sort(pairs 'alphalessp)
  )
)
